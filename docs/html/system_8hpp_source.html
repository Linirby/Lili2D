116) {
117 size_t num_workers = std::thread::hardware_concurrency();
118 size_t chunk_size = entities.size() / num_workers;
119 if (chunk_size == 0) chunk_size = entities.size();
120
121 std::vector<std::future<void>> futures;
122
123 for (size_t start = 0; start < entities.size(); start += chunk_size) {
124 size_t end = std::min(start + chunk_size, entities.size());
125
126 auto promise = std::make_shared<std::promise<void>>();
127 futures.push_back(promise->get_future());

129 pool.enqueue(

141 updateEntity(registry, entities[i], components[i], dt);
142 promise->set_value();
143 },
144 TaskPriority::HIGH
145 );
146 }

148 for (auto& f : futures)
149 f.wait();
150}
